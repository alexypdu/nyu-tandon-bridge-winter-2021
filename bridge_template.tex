	    count = 0;
	    candidate = NULL;
    	
    	for (i = 1; i < nums; i++) {
        	if (count == 0)
        	    candidate = i;
        	if (i == candidate)
        	    count++;
        	else
        	    count =- 1;
    	}
    	
	    return candidate;
	}
